# Function for simualation: Closed-Form GBM, Euler-Murayama, Milstein
def simulate_path(s0, mu, sigma, horizon, timesteps, n_sims,scheme):

    # set the random seed for reproducibility
    random.seed(1)

    # read parameters
    S0 = s0                 # initial spot price
    r = mu                  # mu = rf in risk neutral framework
    T = horizon             # time horizon
    t = timesteps           # number of time steps
    n = n_sims              # number of simulation
    
    # define dt
    dt = T/t                # length of time interval

    # simulate 'n' asset price path with 't' timesteps
    S = zeros((t,n))
    S[0] = S0
      
    
    # Euler-Murayama scheme
    if scheme=='EM':
        for i in range(0, t-1):
            w = random.standard_normal(n)
            S[i+1] = S[i] * (1 + r*dt + sigma*sqrt(dt)*w)
    
    # Milstein scheme
    elif scheme=='Mils':
        for i in range(0, t-1):
            w = random.standard_normal(n)
            #S[i+1] = S[i] * (1 + r*dt + sigma*sqrt(dt)*w)
            S[i+1] = S[i] *(1 + r*dt + sigma*sqrt(dt)*w) + (0.5*sigma*sigma*(w*w-1)*dt)  
    
    # Geometric Brownian Motion
    elif scheme=='GBM':
        for i in range(0, t-1):
            w = random.standard_normal(n)
            S[i+1] = S[i] * e**((r-0.5*sigma*sigma)*dt+(sigma*w*sqrt(dt)))         
                                               
    return S

def asian_avg(df, n, avg, samp ):
    """"
     This function performs discrete or continuous averaging on dataframe by either arithmetic or geometric mean.
     n = number of observation for every simulation. 
         if samp is "cont" then n is obsolete regardless of input (requires a dummy input at least).
     avg: arithmetic -"arith" or geometric- "geo".
     samp: discrete - "disc" or continuous - "cont".
    """      
    #discrete sampling 
    if samp == 'disc':
        #sampling period 
        period  = int(np.round(df.shape[0]/(n)))  
        
        #create a new df containing only values of observed period
        df_p= df.iloc[period::period, :]   
        
        if avg== 'arith':  
            avg_df = df_p.mean(axis=0)
        elif avg== 'geo':
            avg_df = g_mean(df_p)
    
    #continous sampling  
    elif samp == 'cont':
        if avg== 'arith':  
            avg_df = df.mean(axis=0)
        elif avg== 'geo':
            avg_df = g_mean(df)
            
    return avg_df

def asian_payoff(arr, cp , K, avg, samp, avgSorK, n):
     """  
        Purpose: The  function will perform various asian variant payoff upon generated spot path based on user input
        (i) arr: array containing randomized stock path generated with either Euler, Milstein or GBM scheme
        (ii) callput : call or put
        (iii) K, Strike price: will be overriden by the mean if fixed price is specified under avgSorK 
        (iv)  avg: averaging method. arithmetic -"arith" or geometric- "geo"
        (v)   samp: sampling method. continuous- "cont" or discrete- "disc"
        (vi)  avgSorK: avgS means fixed strike, 
                       avgK means the average price will form the strike
        (vii)  n:  number of observation for every simulation. 
                 if samp is "cont" then n is obsolete regardless of input (at least a random number).
    """
    df_arr =  pd.DataFrame(arr)
    asian = asian_avg(df_arr, n, avg, samp)
     
    if cp== 'call':
        if avgSorK == 'avgS':   #Fixed Strike
            payoff = mean(maximum(0, asian - K))
        if avgSorK == 'avgK':   #Fixed Price
            payoff =  mean(maximum(0, arr[-1]-asian ))
                           
    elif cp== 'put':
        if avgSorK == 'avgS':   #Fixed Strike
            payoff = mean(maximum(0, K-asian))
        if avgSorK == 'avgK':   #Fixed Price
            payoff =  mean(maximum(0, asian- arr[-1]))                                
        
    return payoff

READ HERE: HOW TO READ THE TABLES

The table is presented in 2 groups with columns of 4s. Every column corresponds to the subsequent groups' name and value.

The point of such presentation is to have an apple-to-apple relative comparison between EM (relative to GBM) and Milstein (relative to GBM).

Naming convention for each instrument is as follows
(i) Call or Put
(ii) Scheme: Euler, EM/ Milstein, Mils/ Geometric Brownian Motion, GBM 
(iii)Averaging Method: Arithmetic, A/Geometric, G
(iv) Sampling: Continuous, C/ Discrete, D
(v) Strike type: Fixed Prices, FP/ Fixed Strike, FS

Name: (i)_(ii)_(iii)_(iv)_(v), Example: Call_EM_A_C_FS

Instrument name with GBM will not be present as data presented will be relative difference between the schemes against GBM (example Call_GBM_A_C_FS will not be present).
